134class StringPool
135{
136private:
137 char pool[STRING_POOL_COUNT][STRING_POOL_SIZE];
138 bool inUse[STRING_POOL_COUNT];
139 StringAllocation_t m_allocations[STRING_POOL_COUNT];
140 uint8_t m_activeCount;
141 uint32_t m_allocationCount;
142 uint32_t m_deallocationCount;
143
144public:
145 StringPool() : m_activeCount(0), m_allocationCount(0), m_deallocationCount(0)
146 {
147 for (int i = 0; i < STRING_POOL_COUNT; i++)
148 {
149 inUse[i] = false;
150 m_allocations[i].ptr = nullptr;
151 m_allocations[i].isFromPool = false;
152 m_allocations[i].timestamp = 0;
153 m_allocations[i].caller = nullptr;
154 }
155 }
156
157 char* allocate(const char* caller = nullptr)
158 {
159 for (int i = 0; i < STRING_POOL_COUNT; i++)
160 {
161 if (!inUse[i])
162 {
163 inUse[i] = true;
164 char* ptr = pool[i];
165
166 // Track allocation
167 if (m_activeCount < STRING_POOL_COUNT) {
168 m_allocations[m_activeCount].ptr = ptr;
169 m_allocations[m_activeCount].isFromPool = true;
170 m_allocations[m_activeCount].timestamp = millis();
171 m_allocations[m_activeCount].caller = caller;
172 m_activeCount++;
173 }
174 m_allocationCount++;
175
176 return ptr;
177 }
178 }
179 return nullptr; // Pool exhausted
180 }
181
182 void deallocate(char* ptr)
183 {
184 if (ptr)
185 {
186 for (int i = 0; i < STRING_POOL_COUNT; i++)
187 {
188 if (pool[i] == ptr)
189 {
190 inUse[i] = false;
191 pool[i][0] = '\0'; // Clear the string
192
193 // Track deallocation
194 m_deallocationCount++;
195
196 // Remove from tracking
197 for (int j = 0; j < m_activeCount; j++)
198 {
199 if (m_allocations[j].ptr == ptr && m_allocations[j].isFromPool)
200 {
201 // Shift remaining elements left
202 for (int k = j; k < m_activeCount - 1; k++)
203 {
204 m_allocations[k] = m_allocations[k + 1];
205 }
206 m_activeCount--;
207 break;
208 }
209 }
210 return;
211 }
212 }
213 }
214 }

219 void getStats(uint32_t& activeAllocations, uint32_t& totalAllocations, uint32_t& totalDeallocations) const
220 {
221 activeAllocations = m_activeCount;
222 totalAllocations = m_allocationCount;
223 totalDeallocations = m_deallocationCount;
224 }
225
229 const StringAllocation_t* getActiveAllocations() const
230 {
231 return m_allocations;
232 }
233
237 uint8_t getActiveCount() const
238 {
239 return m_activeCount;
240 }
241
245 void cleanup()
246 {
247 for (int i = 0; i < m_activeCount; i++)
248 {
249 if (m_allocations[i].isFromPool)
250 {
251 deallocate(m_allocations[i].ptr);
252 }
253 }
254 m_activeCount = 0;
255 }
