98inline uint16_t hsvToRgb565Fast(float h, float s, float v) {
99 float r, g, b;
100 int i = int(h * 6.0f);
101 float f = h * 6.0f - i;
102 float p = v * (1.0f - s);
103 float q = v * (1.0f - f * s);
104 float t = v * (1.0f - (1.0f - f) * s);
105
106 switch (i % 6) {
107 case 0: r = v; g = t; b = p; break;
108 case 1: r = q; g = v; b = p; break;
109 case 2: r = p; g = v; b = t; break;
110 case 3: r = p; g = q; b = v; break;
111 case 4: r = t; g = p; b = v; break;
112 default: r = v; g = p; b = q; break;
113 }
114
115 uint16_t R = (uint16_t)(r * 31.0f);
116 uint16_t G = (uint16_t)(g * 63.0f);
117 uint16_t B = (uint16_t)(b * 31.0f);
118 return (R << 11) | (G << 5) | B;
119}

126inline HSV_t rgb565ToHsv(uint16_t color565) {
127 // Extrai os canais RGB
128 uint8_t r = (color565 >> 11) & 0x1F; // 5 bits
129 uint8_t g = (color565 >> 5) & 0x3F; // 6 bits
130 uint8_t b = color565 & 0x1F; // 5 bits
131
132 // Converte para 0–1 float
133 float rf = r / 31.0f;
134 float gf = g / 63.0f;
135 float bf = b / 31.0f;
136
137 // Encontra maior e menor
138 float maxVal = max(rf, max(gf, bf));
139 float minVal = min(rf, min(gf, bf));
140 float delta = maxVal - minVal;
141
142 HSV_t hsv;
143 hsv.v = maxVal; // valor = brilho
144
145 // Saturação
146 hsv.s = (maxVal == 0) ? 0 : (delta / maxVal);
147
148 // Matiz
149 if (delta == 0) {
150 hsv.h = 0; // neutro (sem cor)
151 } else if (maxVal == rf) {
152 hsv.h = fmodf(((gf - bf) / delta), 6.0f);
153 } else if (maxVal == gf) {
154 hsv.h = ((bf - rf) / delta) + 2.0f;
155 } else {
156 hsv.h = ((rf - gf) / delta) + 4.0f;
157 }
158
159 hsv.h /= 6.0f;
160 if (hsv.h < 0) hsv.h += 1.0f; // normaliza para [0,1]
161
162 return hsv;
163}

174 inline bool gerarDegradeHSV(float h, int numTons, uint16_t* out, int outSize) {
175 if (numTons <= 0 || out == nullptr || outSize < numTons) return false;
176

179 float v = (numTons > 1) ? (0.3f + (0.7f / (numTons - 1)) * i) : 1.0f;
180 out[i] = hsvToRgb565Fast(h, s, v);

196 inline bool blendColors(uint16_t startColor, uint16_t endColor, int numTons, uint16_t* out, int outSize) {
197 if (numTons < 2 || out == nullptr || outSize < numTons) return false;
198
199 HSV_t a = rgb565ToHsv(startColor);
200 HSV_t b = rgb565ToHsv(endColor);
201
202 if (a.s < 1e-6f) a.h = b.h;
203 if (b.s < 1e-6f) b.h = a.h;
204
205 float dh = b.h - a.h;
206 if (dh > 0.5f) dh -= 1.0f;
207 if (dh < -0.5f) dh += 1.0f;
208
209 for (int i = 0; i < numTons; i++) {
210 float t = (float)i / (numTons - 1);
211
212 float h = a.h + dh * t;
213 if (h < 0.0f) h += 1.0f;
214 if (h >= 1.0f) h -= 1.0f;
215
216 float s = a.s + (b.s - a.s) * t;
217 float v = a.v + (b.v - a.v) * t;
218
219 out[i] = hsvToRgb565Fast(h, s, v);
220 }
221 return true;
222}

234inline bool blendColorsRGB(uint16_t c1, uint16_t c2, int numTons, uint16_t* out, int outSize) {
235 if (numTons < 2 || out == nullptr || outSize < numTons) return false;
236
237 auto extract = [](uint16_t c, uint8_t &r, uint8_t &g, uint8_t &b) {
238 r = ((c >> 11) & 0x1F) << 3;
239 g = ((c >> 5) & 0x3F) << 2;
240 b = (c & 0x1F) << 3;
241 };
242
243 uint8_t r1, g1, b1, r2, g2, b2;
244 extract(c1, r1, g1, b1);

247 for (int i = 0; i < numTons; i++) {
248 float t = (float)i / (numTons - 1);
249
250 uint8_t r = static_cast<uint8_t>(r1 + (r2 - r1) * t);
251 uint8_t g = static_cast<uint8_t>(g1 + (g2 - g1) * t);
252 uint8_t b = static_cast<uint8_t>(b1 + (b2 - b1) * t);
253

255 }

270inline uint16_t lighten565(uint16_t color, float value) {
271 if (value <= 0.0f) return color; // sem alteração
272 if (value >= 1.0f) return 0xFFFF; // branco puro
273
274 // Converte para HSV
275 HSV_t hsv = rgb565ToHsv(color);
276
277 // Interpola o brilho (v) e reduz levemente a saturação
278 hsv.v = hsv.v + (1.0f - hsv.v) * value; // aproxima do branco
279 hsv.s = hsv.s * (1.0f - 0.5f * value); // desatura um pouco ao clarear
280
281 // Converte de volta para RGB565
282 return hsvToRgb565Fast(hsv.h, hsv.s, hsv.v);
283}
